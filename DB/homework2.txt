2.1 Diminuarea salariului cu 10% pentru salariatii unui departament
UPDATE SALARIAT
SET SALARIU = 0.9 * SALARIU
WHERE COD_DEPARTAMENT = (
SELECT DEPARTAMENT.COD_DEPARTAMENT
FROM DEPARTAMENT
WHERE DEPARTAMENT.DENUMIRE_DEPARTAMENT = 'Tehnic');

2.2 Afisarea salariului maxim din fiecare departament
SELECT MAX(SALARIU)
FROM SALARIAT
GROUP BY COD_DEPARTAMENT;

2.3 Afisarea salariului maxim pentru fiecare departament care incepe cu litera T si are peste 30 salariati.
SELECT MAX(S.SALARIU)
FROM SALARIAT S 
INNER JOIN DEPARTAMENT D
ON S.COD_DEPARTAMENT = D.COD_DEPARTAMENT
WHERE D.DENUMIRE_DEPARTAMENT LIKE 'T%'
GROUP BY S.COD_DEPARTAMENT
HAVING COUNT(S.COD_DEPARTAMENT) >= 30;

2.4 Majorarea salariului cu 10% pentru salariatii incadrati in departamente care au un singur salariat.
UPDATE SALARIAT
SET SALARIU=1.1*SALARIU
WHERE COD_DEPARTAMENT IN (
SELECT S.COD_DEPARTAMENT
FROM SALARIAT S
GROUP BY S.COD_DEPARTAMENT
HAVING COUNT(*)=1);
